local function deleteStopCheatingBum(instance)
    if instance.Name == "stopcheatingbum" then

local function monitorAvailable(folder)
    if not folder then return end
    
    local existing = folder:FindFirstChild("stopcheatingbum")
    if existing then
        existing:Destroy()
        warn("bypassed")
    end
    
    folder.ChildAdded:Connect(deleteStopCheatingBum)
end

local function setupWatcher()
    local tryoutCache = replicatedStorage:FindFirstChild("CURRENT_TRYOUT_CACHE")
    
    if not tryoutCache then
        replicatedStorage.ChildAdded:Connect(function(child)
            if child.Name == "CURRENT_TRYOUT_CACHE" then
                local available = child:FindFirstChild("Available")
                if available then
                    monitorAvailable(available)
                else
                    child.ChildAdded:Connect(function(subChild)
                        if subChild.Name == "Available" then
                            monitorAvailable(subChild)
                        end
                    end)
                end
            end
        end)
    else
        local available = tryoutCache:FindFirstChild("Available")
        if available then
            monitorAvailable(available)
        else
            tryoutCache.ChildAdded:Connect(function(child)
                if child.Name == "Available" then
                    monitorAvailable(child)
                end
            end)
        end
    end
end

local desync_api = {}
do
    local enabled, hooked, get_root, target_cf, real_cf, hrp, old_idx = true, false
    get_root = function()
        local char = getgenv().LocalPlayer.Character
        if not char then return nil end
        local hum = char:FindFirstChildOfClass("Humanoid")
        return hum and hum.RootPart
    end

    old_idx = hookmetamethod(game, "__index", newcclosure(function(self, key)
        if not checkcaller() and key == "CFrame" and enabled and self == hrp and real_cf then
            return real_cf
        end
        return old_idx(self, key)
    end))
    hooked = true

    local desync_conn = getgenv().RunService.Heartbeat:Connect(function()
        if not enabled then return end
        hrp = get_root()
        if not hrp then return end
        real_cf = hrp.CFrame
        hrp.CFrame = target_cf or real_cf
        getgenv().RunService.RenderStepped:Wait()
        hrp.CFrame = real_cf
    end)

    desync_api.move_desync = function(pos)
        local t = typeof(pos)
        if t == "CFrame" then
            target_cf = pos
        elseif t == "Vector3" then
            target_cf = CFrame.new(pos)
        else
            return nil
        end
        return target_cf
    end

    desync_api.sync_desync = function()
        target_cf = nil
        return hrp and hrp.CFrame
    end

    desync_api.get_desync_pos = function()
        return target_cf and target_cf.Position or (hrp and hrp.Position)
    end

    desync_api.get_raw_pos = function()
        return real_cf and real_cf.Position
    end
end

getgenv().updateHighlight = function(character, color)
    if not character or not character.Parent then
        if getgenv().CurrentHighlight then
            getgenv().CurrentHighlight:Destroy()
            getgenv().CurrentHighlight = nil
        end
        return
    end
    if not getgenv().CurrentHighlight then
        getgenv().CurrentHighlight = Instance.new("Highlight")
        getgenv().CurrentHighlight.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        getgenv().CurrentHighlight.FillTransparency = 0.5
        getgenv().CurrentHighlight.OutlineTransparency = 0.7
    end
    getgenv().CurrentHighlight.FillColor = color
    getgenv().CurrentHighlight.OutlineColor = color
    getgenv().CurrentHighlight.Parent = character
end

getgenv().clearHighlight = function()
    if getgenv().CurrentHighlight then
        getgenv().CurrentHighlight:Destroy()
        getgenv().CurrentHighlight = nil
    end
    getgenv().CurrentTargetCharacter = nil
end

getgenv().isVisible = function(part)
    if not part or not part.Parent then return false end
    local origin = getgenv().Camera.CFrame.Position
    local direction = part.Position - origin
    local params = RaycastParams.new()
    params.FilterDescendantsInstances = {getgenv().LocalPlayer.Character}
    params.FilterType = Enum.RaycastFilterType.Blacklist
    local ray = getgenv().Workspace:Raycast(origin, direction, params)
    if not ray then return true end
    local hit = ray.Instance
    if hit and hit:IsDescendantOf(part.Parent) then return true end
    return false
end

getgenv().passesChecks = function(character)
    if not character then return false end
    local bodyEffects = character:FindFirstChild("BodyEffects")
    if getgenv().Silent.KOCheck and bodyEffects then
        local ko = bodyEffects:FindFirstChild("K.O")
        if ko and ko.Value then return false end
    end
    if getgenv().Silent.GrabCheck and character:FindFirstChild("GRABBING_CONSTRAINT", true) then
        return false
    end
    if getgenv().Silent.WallCheck then
        local part = character:FindFirstChild(getgenv().Silent.TargetPart)
        if not part or not getgenv().isVisible(part) then
            return false
        end
    end
    return true
end

getgenv().getClosestTarget = function()
    if getgenv().Sticky.Enabled and getgenv().Sticky.LockedPlayer then
        local char = getgenv().Sticky.LockedPlayer.Character
        if char and char.Parent and getgenv().passesChecks(char) then
            local part = char:FindFirstChild(getgenv().Silent.TargetPart)
            if part then
                getgenv().Sticky.CurrentTarget = part
                return part
            end
        end
        if not getgenv().Sticky.LockedPlayer.Parent then
            getgenv().Sticky.LockedPlayer = nil
            getgenv().Sticky.CurrentTarget = nil
        end
        return getgenv().Sticky.CurrentTarget
    end
    local closestPart
    local closestPlayer
    local closestDist = getgenv().Silent.UseFOV and getgenv().Silent.FOVRadius or math.huge
    local reference = getgenv().Silent.FOVPlacement == "center" and Vector2.new(getgenv().Camera.ViewportSize.X / 2, getgenv().Camera.ViewportSize.Y / 2) or getgenv().UserInputService:GetMouseLocation()
    for _, plr in ipairs(getgenv().Players:GetPlayers()) do
        if plr ~= getgenv().LocalPlayer and plr.Character and getgenv().passesChecks(plr.Character) then
            local part = plr.Character:FindFirstChild(getgenv().Silent.TargetPart)
            if part then
                local pos, onscreen = getgenv().Camera:WorldToViewportPoint(part.Position)
                if onscreen then
                    local dist = (Vector2.new(pos.X, pos.Y) - reference).Magnitude
                    if dist < closestDist then
                        closestDist = dist
                        closestPart = part
                        closestPlayer = plr
                    end
                end
            end
        end
    end
    if getgenv().Sticky.Enabled and closestPlayer then
        getgenv().Sticky.LockedPlayer = closestPlayer
        getgenv().Sticky.CurrentTarget = closestPart
    end
    return closestPart
end

getgenv().RunService.RenderStepped:Connect(function()
    if not getgenv().Silent.Enabled or not getgenv().Silent.UseFOV then
        getgenv().FOVCircle.Visible = false
        getgenv().FOVInnerOutline.Visible = false
        getgenv().FOVOuterOutline.Visible = false
    else
        local mousePos = getgenv().UserInputService:GetMouseLocation()
        local center = Vector2.new(getgenv().Camera.ViewportSize.X / 2, getgenv().Camera.ViewportSize.Y / 2)
        local pos = getgenv().Silent.FOVPlacement == "center" and center or mousePos
        getgenv().FOVCircle.Position = pos
        getgenv().FOVInnerOutline.Position = pos
        getgenv().FOVOuterOutline.Position = pos
        getgenv().FOVCircle.Radius = getgenv().Silent.FOVRadius
        getgenv().FOVInnerOutline.Radius = getgenv().Silent.FOVRadius - 2
        getgenv().FOVOuterOutline.Radius = getgenv().Silent.FOVRadius + 2
        getgenv().FOVCircle.Color = getgenv().Silent.FOVColor
        getgenv().FOVCircle.Transparency = getgenv().Silent.FOVTransparency
        getgenv().FOVCircle.Visible = true
        if getgenv().Silent.OutlineEnabled then
            getgenv().FOVInnerOutline.Visible = true
            getgenv().FOVOuterOutline.Visible = true
        else
            getgenv().FOVInnerOutline.Visible = false
            getgenv().FOVOuterOutline.Visible = false
        end
    end
    if getgenv().Camlock.Enabled and getgenv().Camlock.UseFOV then
        local mousePos = getgenv().UserInputService:GetMouseLocation()
        local center = Vector2.new(getgenv().Camera.ViewportSize.X / 2, getgenv().Camera.ViewportSize.Y / 2)
        local pos = getgenv().Camlock.FOVPlacement == "center" and center or mousePos
        getgenv().CamlockFOVCircle.Position = pos
        getgenv().CamlockInnerOutline.Position = pos
        getgenv().CamlockOuterOutline.Position = pos
        getgenv().CamlockFOVCircle.Radius = getgenv().Camlock.FOVRadius
        getgenv().CamlockInnerOutline.Radius = getgenv().Camlock.FOVRadius - 2
        getgenv().CamlockOuterOutline.Radius = getgenv().Camlock.FOVRadius + 2
        getgenv().CamlockFOVCircle.Color = getgenv().Camlock.FOVColor
        getgenv().CamlockFOVCircle.Transparency = getgenv().Camlock.FOVTransparency
        getgenv().CamlockFOVCircle.Visible = true
        if getgenv().Camlock.OutlineEnabled then
            getgenv().CamlockInnerOutline.Visible = true
            getgenv().CamlockOuterOutline.Visible = true
        else
            getgenv().CamlockInnerOutline.Visible = false
            getgenv().CamlockOuterOutline.Visible = false
        end
    else
        getgenv().CamlockFOVCircle.Visible = false
        getgenv().CamlockInnerOutline.Visible = false
        getgenv().CamlockOuterOutline.Visible = false
    end
    local camlockTargetPart = nil
    if getgenv().Camlock.Enabled then
        camlockTargetPart = getgenv().getCamlockTarget()
        if camlockTargetPart then
            local predictedPos = camlockTargetPart.Position + (camlockTargetPart.Velocity * getgenv().Camlock.Prediction)
            local targetCFrame = CFrame.new(getgenv().Camera.CFrame.Position, predictedPos)
            getgenv().Camera.CFrame = getgenv().Camera.CFrame:Lerp(targetCFrame, getgenv().Camlock.Smoothness)
        end
    end
    local silentTargetPart = nil
    if getgenv().Silent.Enabled then
        silentTargetPart = getgenv().getClosestTarget()
    end
    local activeTargetPart = nil
    local activeTargetCharacter = nil
    if getgenv().Camlock.Enabled and camlockTargetPart then
        activeTargetPart = camlockTargetPart
        activeTargetCharacter = camlockTargetPart.Parent
    elseif getgenv().Silent.Enabled and silentTargetPart then
        activeTargetPart = silentTargetPart
        activeTargetCharacter = silentTargetPart.Parent
    end
    local tracerEnabled = library.flags.tracertgl
    if tracerEnabled and activeTargetPart and activeTargetCharacter then
        local mousePos = getgenv().UserInputService:GetMouseLocation()
        local targetPos, onScreen = getgenv().Camera:WorldToViewportPoint(activeTargetPart.Position)
        if onScreen then
            local targetVec = Vector2.new(targetPos.X, targetPos.Y)
            getgenv().TracerOutline.From = mousePos
            getgenv().TracerOutline.To = targetVec
            getgenv().TracerFill.From = mousePos
            getgenv().TracerFill.To = targetVec
            getgenv().TracerOutline.Visible = true
            getgenv().TracerFill.Visible = true
        else
            getgenv().TracerOutline.Visible = false
            getgenv().TracerFill.Visible = false
        end
    else
        getgenv().TracerOutline.Visible = false
        getgenv().TracerFill.Visible = false
    end
    local highlightEnabled = library.flags.highlighttgl
    if highlightEnabled then
        if activeTargetCharacter then
            local highlightColor = library.flags.highlightcolor and library.flags.highlightcolor.Color or Color3.fromRGB(255, 255, 255)
            if activeTargetCharacter ~= getgenv().CurrentTargetCharacter then
                getgenv().clearHighlight()
                getgenv().updateHighlight(activeTargetCharacter, highlightColor)
                getgenv().CurrentTargetCharacter = activeTargetCharacter
            elseif getgenv().CurrentHighlight then
                getgenv().CurrentHighlight.FillColor = highlightColor
                getgenv().CurrentHighlight.OutlineColor = highlightColor
            end
        else
            getgenv().clearHighlight()
        end
    else
        getgenv().clearHighlight()
    end
end)

_G.triggerbot = false
_G.triggerbotKey = Enum.KeyCode.E
_G.triggerbotDelay = 0.1
_G.triggerbotWhitelist = {
    head = true,
    humanoid = true,
    torso = true,
    ["upper torso"] = true,
    ["lower torso"] = true,
    ["left arm"] = true,
    ["right arm"] = true,
    ["left leg"] = true,
    ["right leg"] = true,
}

local player = getgenv().LocalPlayer
local mouse = player:GetMouse()
local lastShoot = 0

getgenv().isPartWhitelisted = function(part)
    if not part then return false end
    
    local partName = part.Name
    for whitelistPart, enabled in pairs(_G.triggerbotWhitelist) do
        if enabled then
            local targetPart = getgenv().hitPartMap[whitelistPart] or whitelistPart
            if partName == targetPart or partName:lower():find(whitelistPart:lower()) then
                return true
            end
        end
    end
    return false
end

getgenv().isValidTarget = function(part)
    if not part then return false end
    
    local character = part.Parent
    if not character then return false end
    
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not humanoid or humanoid.Health <= 0 then return false end
    
    local playerFromChar = getgenv().Players:GetPlayerFromCharacter(character)
    if not playerFromChar or playerFromChar == player then return false end
    
    return true
end

getgenv().triggerbotShoot = function()
    local tool = player.Character and player.Character:FindFirstChildOfClass("Tool")
    if tool then
        tool:Activate()
    end
end

local triggerbotConnection = nil

getgenv().startTriggerbotLoop = function()
    if triggerbotConnection then return end
    
    triggerbotConnection = getgenv().RunService.RenderStepped:Connect(function()
        if not _G.triggerbot then return end
        
        local target = mouse.Target
        if not target then return end
        
        local isWhitelisted = getgenv().isPartWhitelisted(target)
        local isValid = getgenv().isValidTarget(target)
        
        if isValid and isWhitelisted then
            if tick() - lastShoot >= _G.triggerbotDelay then
                lastShoot = tick()
                getgenv().triggerbotShoot()
            end
        end
    end)
end

getgenv().stopTriggerbotLoop = function()
    if triggerbotConnection then
        triggerbotConnection:Disconnect()
        triggerbotConnection = nil
    end
end

getgenv().forcehitCanShoot = function(player)
    if not player or not player.Character then return false end
    local hum = player.Character:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    if getgenv().forcehitSettings.koCheck then
        local bodyEffects = player.Character:FindFirstChild("BodyEffects")
        if bodyEffects then
            local ko = bodyEffects:FindFirstChild("K.O")
            if ko and ko.Value then return false end
        end
    end
    if getgenv().forcehitSettings.forcefieldCheck then
        if player.Character:FindFirstChildOfClass("ForceField") then
            return false
        end
    end
    return true
end

getgenv().forcehitTargetValid = function(player)
    if not player or player == getgenv().LocalPlayer then return false end
    if not player.Character then return false end
    local hum = player.Character:FindFirstChildOfClass("Humanoid")
    if not hum or hum.Health <= 0 then return false end
    return true
end

getgenv().forcehitGetCurrentTarget = function()
    if getgenv().forcehitSettings.lockTarget then
        local locked = getgenv().forcehitGetLockedTarget()
        if locked then
            return locked
        end
    end
    if getgenv().forcehitSettings.enabled then
        return getgenv().forcehitTarget
    end
    return nil
end

getgenv().forcehitShouldShoot = function()
    return getgenv().forcehitSettings.enabled or getgenv().forcehitSettings.lockTarget
end

getgenv().forcehitShoot = function()
    if not getgenv().forcehitShouldShoot() then return end
    local target = getgenv().forcehitGetCurrentTarget()
    if not target or not getgenv().forcehitCanShoot(target) then return end
    local mainEvent = getgenv().getMainEvent()
    if not mainEvent then return end
    local head = target.Character:FindFirstChild("Head")
    local hrp = target.Character:FindFirstChild("HumanoidRootPart")
    local targetPart = head or hrp
    if not targetPart then return end
    local myHrp = getgenv().LocalPlayer.Character and getgenv().LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    local myPos = myHrp and myHrp.Position or Vector3.zero
    local args = {
        "Shoot",
        {
            {
                { Normal = targetPart.Position, Instance = targetPart, Position = targetPart.Position },
                { Normal = targetPart.Position, Instance = targetPart, Position = targetPart.Position },
                { Normal = targetPart.Position, Instance = targetPart, Position = targetPart.Position },
                { Normal = targetPart.Position, Instance = targetPart, Position = targetPart.Position },
                { Normal = targetPart.Position, Instance = targetPart, Position = targetPart.Position }
            },
            {
                { thePart = targetPart, theOffset = Vector3.zero },
                { thePart = targetPart, theOffset = Vector3.zero },
                { thePart = targetPart, theOffset = Vector3.zero },
                { thePart = targetPart, theOffset = Vector3.zero },
                { thePart = targetPart, theOffset = Vector3.zero }
            },
            myPos,
            myPos,
            workspace:GetServerTimeNow()
        }
    }
    pcall(function()
        mainEvent:FireServer(unpack(args))
    end)
end

getgenv().forcehitLoop = getgenv().RunService.Heartbeat:Connect(function()
    local lockedTarget = getgenv().forcehitGetLockedTarget()

    if getgenv().forcehitSettings.lockTarget and lockedTarget then
        if not getgenv().forcehitTargetValid(lockedTarget) then
            getgenv().forcehitSetLockedTarget(nil)
            lockedTarget = nil
        end
    end

    if not getgenv().forcehitSettings.lockTarget and getgenv().forcehitSettings.enabled then
        local closestPlayer = nil
        local closestDist = math.huge
        local mousePos = getgenv().UserInputService:GetMouseLocation()

        for _, plr in ipairs(getgenv().Players:GetPlayers()) do
            if getgenv().forcehitTargetValid(plr) then
                local hrp = plr.Character:FindFirstChild("HumanoidRootPart")
                if hrp then
                    local pos, onScreen = getgenv().Camera:WorldToViewportPoint(hrp.Position)
                    if onScreen then
                        local dist = (Vector2.new(pos.X, pos.Y) - mousePos).Magnitude
                        if dist < closestDist then
                            closestDist = dist
                            closestPlayer = plr
                        end
                    end
                end
            end
        end

        getgenv().forcehitTarget = closestPlayer
    end

    if getgenv().forcehitSettings.highlight then
        getgenv().forcehitUpdateHighlight()
    elseif getgenv().forcehitHighlight then
        getgenv().forcehitHighlight:Destroy()
        getgenv().forcehitHighlight = nil
    end

    local currentTarget = getgenv().forcehitGetCurrentTarget()
    if getgenv().forcehitSettings.tracer and currentTarget and currentTarget.Character then
        local hrp = currentTarget.Character:FindFirstChild("HumanoidRootPart")
        if hrp then
            local targetPos, onScreen = getgenv().Camera:WorldToViewportPoint(hrp.Position)
            if onScreen then
                if not getgenv().forcehitTracerOutline then
                    getgenv().forcehitTracerOutline = Drawing.new("Line")
                    getgenv().forcehitTracerOutline.Thickness = 2
                    getgenv().forcehitTracerOutline.Color = Color3.fromRGB(0, 0, 0)
                    getgenv().forcehitTracerOutline.Visible = true
                    getgenv().forcehitTracerMain = Drawing.new("Line")
                    getgenv().forcehitTracerMain.Thickness = 1
                    getgenv().forcehitTracerMain.Visible = true
                end
                local mousePos = getgenv().UserInputService:GetMouseLocation()
                local targetVec = Vector2.new(targetPos.X, targetPos.Y)
                getgenv().forcehitTracerOutline.From = mousePos
                getgenv().forcehitTracerOutline.To = targetVec
                getgenv().forcehitTracerMain.From = mousePos
                getgenv().forcehitTracerMain.To = targetVec
                getgenv().forcehitTracerMain.Color = getgenv().forcehitSettings.tracerColor
                getgenv().forcehitTracerOutline.Visible = true
                getgenv().forcehitTracerMain.Visible = true
            else
                if getgenv().forcehitTracerOutline then
                    getgenv().forcehitTracerOutline.Visible = false
                    getgenv().forcehitTracerMain.Visible = false
                end
            end
        end
    else
        if getgenv().forcehitTracerOutline then
            getgenv().forcehitTracerOutline.Visible = false
            getgenv().forcehitTracerMain.Visible = false
        end
    end
end)

if getgenv().forcehitShouldShoot() then
    getgenv().startAutoShoot()
end

local ESPObjects = {}

local ToolObjects = {}
local DistObjects = {}
local HealthbarObjects = {}
local NametagHooks = {}

local PartsCache = {}
local PartsCacheDirty = {}
local CharacterConnections = {}

local NAME_BASE_SIZE = 13
local NAME_BASE_DIST = 40
local NAME_MIN_SIZE = 11
local NAME_MAX_SIZE = 13

local FIXED_HW = 1.98
local FIXED_HH = 2.94
local FIXED_OY = -0.01

local HB_WIDTH = 1
local HB_OFFSET = 5

local flags = library.flags

getgenv().SetHumanoidDisplay = function(hum, hide)
    if hum then
        hum.DisplayDistanceType = hide and Enum.HumanoidDisplayDistanceType.None or Enum.HumanoidDisplayDistanceType.Viewer
    end
end

getgenv().ApplyNametag = function(player, hide)
    if player == getgenv().LocalPlayer then return end
    if player.Character then
        local hum = player.Character:FindFirstChildOfClass("Humanoid")
        getgenv().SetHumanoidDisplay(hum, hide)
    end
end

getgenv().HookPlayerNametag = function(player)
    if player == getgenv().LocalPlayer then return end
    if NametagHooks[player] then NametagHooks[player]:Disconnect() end
    NametagHooks[player] = player.CharacterAdded:Connect(function(char)
        local hum = char:WaitForChild("Humanoid", 5)
        if hum then getgenv().SetHumanoidDisplay(hum, flags["hiderobloxnames"]) end
    end)
end

getgenv().UpdateAllNametags = function()
    local hide = flags["hiderobloxnames"] or false
    for _, player in ipairs(getgenv().Players:GetPlayers()) do
        getgenv().ApplyNametag(player, hide)
    end
end

getgenv().UpdateAllNametags()

local lastHideState = flags["hiderobloxnames"]
getgenv().RunService.Heartbeat:Connect(function()
    if flags["hiderobloxnames"] ~= lastHideState then
        lastHideState = flags["hiderobloxnames"]
        getgenv().UpdateAllNametags()
    end
end)


getgenv().BuildPartsCache = function(player)
    local char = player.Character
    if not char then
        PartsCache[player] = { all = {}, accs = {} }
        return
    end
    local parts = { all = {}, accs = {} }
    for _, v in ipairs(char:GetDescendants()) do
        if v:IsA("BasePart") then
            local ancestor = v.Parent
            local inTool = false
            while ancestor and ancestor ~= char do
                if ancestor:IsA("Tool") then inTool = true break end
                ancestor = ancestor.Parent
            end
            if inTool then continue end
            local parent = v.Parent
            local isAcc = parent:IsA("Accessory") or parent:IsA("Hat")
            if isAcc then
                table.insert(parts.accs, v)
            else
                table.insert(parts.all, v)
            end
        end
    end
    PartsCache[player] = parts
    PartsCacheDirty[player] = false
end

getgenv().HookCharacterForCache = function(player)
    if CharacterConnections[player] then
        for _, conn in ipairs(CharacterConnections[player]) do conn:Disconnect() end
    end
    CharacterConnections[player] = {}
    local char = player.Character
    if not char then return end
    table.insert(CharacterConnections[player], char.ChildAdded:Connect(function(child)
        PartsCacheDirty[player] = true
        if child:IsA("Accessory") or child:IsA("Hat") then
            child.ChildAdded:Connect(function() PartsCacheDirty[player] = true end)
        end
    end))
    table.insert(CharacterConnections[player], char.ChildRemoved:Connect(function()
        PartsCacheDirty[player] = true
    end))
end

local _cbuf = table.create(8)

getgenv().GetAllCorners = function(part)
    local cf = part.CFrame
    local sx, sy, sz = part.Size.X * 0.5, part.Size.Y * 0.5, part.Size.Z * 0.5
    _cbuf[1] = (cf * CFrame.new( sx,  sy,  sz)).Position
    _cbuf[2] = (cf * CFrame.new(-sx,  sy,  sz)).Position
    _cbuf[3] = (cf * CFrame.new( sx, -sy,  sz)).Position
    _cbuf[4] = (cf * CFrame.new(-sx, -sy,  sz)).Position
    _cbuf[5] = (cf * CFrame.new( sx,  sy, -sz)).Position
    _cbuf[6] = (cf * CFrame.new(-sx,  sy, -sz)).Position
    _cbuf[7] = (cf * CFrame.new( sx, -sy, -sz)).Position
    _cbuf[8] = (cf * CFrame.new(-sx, -sy, -sz)).Position
    return _cbuf
end

getgenv().GetFixedCorners = function(rootPos)
    local hw = FIXED_HW
    local hh = FIXED_HH
    local cy = rootPos.Y + FIXED_OY
    local x, z = rootPos.X, rootPos.Z
    _cbuf[1] = Vector3.new(x - hw, cy + hh, z)
    _cbuf[2] = Vector3.new(x + hw, cy + hh, z)
    _cbuf[3] = Vector3.new(x - hw, cy - hh, z)
    _cbuf[4] = Vector3.new(x + hw, cy - hh, z)
    _cbuf[5] = Vector3.new(x,      cy + hh, z - hw)
    _cbuf[6] = Vector3.new(x,      cy + hh, z + hw)
    _cbuf[7] = Vector3.new(x,      cy - hh, z - hw)
    _cbuf[8] = Vector3.new(x,      cy - hh, z + hw)
    return _cbuf
end

getgenv().CreateBox = function()
    local fill = Drawing.new("Square")
    fill.Filled = true
    fill.Thickness = 0
    fill.Visible = false
    fill.ZIndex = 1

    local sqOut = Drawing.new("Square")
    sqOut.Filled = false
    sqOut.Thickness = 1
    sqOut.Visible = false
    sqOut.ZIndex = 2

    local sq = Drawing.new("Square")
    sq.Filled = false
    sq.Thickness = 1
    sq.Visible = false
    sq.ZIndex = 3

    local sqIn = Drawing.new("Square")
    sqIn.Filled = false
    sqIn.Thickness = 1
    sqIn.Visible = false
    sqIn.ZIndex = 4

    local lines, outlines = {}, {}
    for i = 1, 8 do
        local ol = Drawing.new("Line")
        ol.Thickness = 2
        ol.Visible = false
        ol.ZIndex = 2
        outlines[i] = ol

        local ln = Drawing.new("Line")
        ln.Thickness = 1
        ln.Visible = false
        ln.ZIndex = 3
        lines[i] = ln
    end

    return { lines = lines, outlines = outlines, fill = fill, sq = sq, sqOut = sqOut, sqIn = sqIn }
end

getgenv().CreateHealthbarDrawings = function()
    local bar = Drawing.new("Square")
    bar.Filled = true bar.Thickness = 0 bar.Visible = false bar.ZIndex = 3

    local outline = Drawing.new("Square")
    outline.Filled = false outline.Thickness = 1 outline.Visible = false outline.ZIndex = 4

    return { bar = bar, outline = outline }
end

getgenv().RemoveBox = function(box)
    box.fill:Remove()
    box.sq:Remove()
    box.sqOut:Remove()
    box.sqIn:Remove()
    for i = 1, 8 do box.lines[i]:Remove() box.outlines[i]:Remove() end
end

getgenv().HideBox = function(box)
    box.fill.Visible = false
    box.sq.Visible = false
    box.sqOut.Visible = false
    box.sqIn.Visible = false
    for i = 1, 8 do box.lines[i].Visible = false box.outlines[i].Visible = false end
end

getgenv().RemoveHealthbarDrawings = function(hb)
    hb.bar:Remove() hb.outline:Remove()
end

getgenv().HideHealthbarDrawings = function(hb)
    hb.bar.Visible = false hb.outline.Visible = false
end

getgenv().HideAll = function(player)
    local box   = ESPObjects[player]
    local label = NameObjects[player]
    local tlabel = ToolObjects[player]
    local dlabel = DistObjects[player]
    local hb    = HealthbarObjects[player]
    if box    then getgenv().HideBox(box) end
    if label  then label.Visible  = false end
    if tlabel then tlabel.Visible = false end
    if dlabel then dlabel.Visible = false end
    if hb     then getgenv().HideHealthbarDrawings(hb) end
end

getgenv().SetFill = function(box, minX, minY, maxX, maxY, fc, ft, sf)
    if not sf then
        box.fill.Visible = false
        return
    end
    box.fill.Position = Vector2.new(minX, minY)
    box.fill.Size = Vector2.new(maxX - minX, maxY - minY)
    box.fill.Color = fc
    box.fill.Transparency = 1 - ft
    box.fill.Visible = true
end

getgenv().Draw2D = function(box, minX, minY, maxX, maxY, bc, oc, sb, so, fc, ft, sf)
    for i = 1, 8 do
        box.lines[i].Visible = false
        box.outlines[i].Visible = false
    end

    local w = math.max(maxX - minX, 1)
    local h = math.max(maxY - minY, 1)

    if sb then
        if so then
            box.sqOut.Position = Vector2.new(minX - 1, minY - 1)
            box.sqOut.Size = Vector2.new(w + 2, h + 2)
            box.sqOut.Color = oc
            box.sqOut.Visible = true
        else
            box.sqOut.Visible = false
        end

        box.sq.Position = Vector2.new(minX, minY)
        box.sq.Size = Vector2.new(w, h)
        box.sq.Color = bc
        box.sq.Visible = true

        if so then
            box.sqIn.Position = Vector2.new(minX + 1, minY + 1)
            box.sqIn.Size = Vector2.new(math.max(w - 2, 1), math.max(h - 2, 1))
            box.sqIn.Color = oc
            box.sqIn.Visible = true
        else
            box.sqIn.Visible = false
        end
    else
        box.sq.Visible = false
        box.sqOut.Visible = false
        box.sqIn.Visible = false
    end

    getgenv().SetFill(box, minX, minY, maxX, maxY, fc, ft, sf)
end

getgenv().SetLine = function(box, idx, a, b, bc, oc, sb, so)
    box.outlines[idx].From = a  box.outlines[idx].To = b
    box.outlines[idx].Color = oc  box.outlines[idx].Visible = so
    box.lines[idx].From = a  box.lines[idx].To = b
    box.lines[idx].Color = bc  box.lines[idx].Visible = sb
end

getgenv().DrawCorner = function(box, minX, minY, maxX, maxY, bc, oc, sb, so, fc, ft, sf)
    box.sq.Visible    = false
    box.sqOut.Visible = false
    box.sqIn.Visible  = false

    local cw = (maxX - minX) * 0.22
    local ch = (maxY - minY) * 0.22
    local segs = {
        { Vector2.new(minX, minY), Vector2.new(minX + cw, minY) },
        { Vector2.new(minX, minY), Vector2.new(minX, minY + ch) },
        { Vector2.new(maxX, minY), Vector2.new(maxX - cw, minY) },
        { Vector2.new(maxX, minY), Vector2.new(maxX, minY + ch) },
        { Vector2.new(minX, maxY), Vector2.new(minX + cw, maxY) },
        { Vector2.new(minX, maxY), Vector2.new(minX, maxY - ch) },
        { Vector2.new(maxX, maxY), Vector2.new(maxX - cw, maxY) },
        { Vector2.new(maxX, maxY), Vector2.new(maxX, maxY - ch) },
    }
    for i, seg in ipairs(segs) do
        getgenv().SetLine(box, i, seg[1], seg[2], bc, oc, sb, so)
    end

    getgenv().SetFill(box, minX, minY, maxX, maxY, fc, ft, sf)
end

getgenv().AddPlayer = function(player, forceSelf)
    if player == getgenv().LocalPlayer and not forceSelf then return end
    if ESPObjects[player] then return end
    ESPObjects[player]       = getgenv().CreateBox()
    NameObjects[player]      = getgenv().CreateTopLabel()
    ToolObjects[player]      = getgenv().CreateBottomLabel()
    DistObjects[player]      = getgenv().CreateBottomLabel()
    HealthbarObjects[player] = getgenv().CreateHealthbarDrawings()
    getgenv().HookPlayerNametag(player)
    getgenv().ApplyNametag(player, flags["hiderobloxnames"])
    PartsCacheDirty[player] = true
    PartsCache[player] = { all = {}, accs = {} }

    if player.Character then
        getgenv().HookCharacterForCache(player)
        getgenv().BuildPartsCache(player)
    end

    if CharacterConnections[player] == nil then CharacterConnections[player] = {} end
    table.insert(CharacterConnections[player], player.CharacterAdded:Connect(function()
        PartsCacheDirty[player] = true
        PartsCache[player] = { all = {}, accs = {} }
        task.defer(function() getgenv().HookCharacterForCache(player) end)
    end))
end

getgenv().RemovePlayer = function(player)
    if ESPObjects[player] then getgenv().RemoveBox(ESPObjects[player]) ESPObjects[player] = nil end
    if NameObjects[player] then NameObjects[player]:Destroy() NameObjects[player] = nil end
    if ToolObjects[player] then ToolObjects[player]:Destroy() ToolObjects[player] = nil end
    if DistObjects[player] then DistObjects[player]:Destroy() DistObjects[player] = nil end
    if HealthbarObjects[player] then getgenv().RemoveHealthbarDrawings(HealthbarObjects[player]) HealthbarObjects[player] = nil end
    if NametagHooks[player] then NametagHooks[player]:Disconnect() NametagHooks[player] = nil end
    if CharacterConnections[player] then
        for _, conn in ipairs(CharacterConnections[player]) do conn:Disconnect() end
        CharacterConnections[player] = nil
    end
    PartsCache[player] = nil
    PartsCacheDirty[player] = nil
end

for _, player in ipairs(getgenv().Players:GetPlayers()) do getgenv().AddPlayer(player) end
getgenv().Players.PlayerAdded:Connect(getgenv().AddPlayer)
getgenv().Players.PlayerRemoving:Connect(getgenv().RemovePlayer)

local lastInclAcc = flags["includeaccessories"]

getgenv().RunService.RenderStepped:Connect(function()
    if not flags["enableesp"] then
        for player in pairs(ESPObjects) do getgenv().HideAll(player) end
        return
    end

    local showBox     = flags["boxes"] or false
    local showOutline = showBox and (flags["boxesoutline"] or false)
    local showFill    = showBox and (flags["boxesfill"] or false)
    local boxColor    = flags["boxescolor"]      and flags["boxescolor"].Color      or Color3.fromRGB(255, 255, 255)
    local outColor    = flags["boxescoloroutline"] and flags["boxescoloroutline"].Color or Color3.fromRGB(0, 0, 0)
    local fillColor   = flags["boxesfillcolor"]  and flags["boxesfillcolor"].Color  or Color3.fromRGB(255, 255, 255)
    local fillTrans   = flags["boxesfillcolor"]  and flags["boxesfillcolor"].Transparency or 0.3
    local mode        = flags["boxplacement"] or "2d"
    local sizeType    = flags["boxsizetype"]  or "dynamic"
    local inclAcc     = flags["includeaccessories"] or true

    local showNames   = flags["names"] or false
    local nameColor   = flags["namescolor"]       and flags["namescolor"].Color       or Color3.fromRGB(255, 255, 255)
    local showNOutline = showNames and (flags["namesoutline"] or false)
    local nOutColor   = flags["namescoloroutline"] and flags["namescoloroutline"].Color or Color3.fromRGB(0, 0, 0)
    local nameFont    = fontMap[flags["namefont"] or "gotham"] or Enum.Font.Gotham
    local nameType    = flags["nametype"] or "displayname"

    local showTools   = flags["tools"] or false
    local toolColor   = flags["toolcolor"]       and flags["toolcolor"].Color       or Color3.fromRGB(255, 255, 255)
    local showTOutline = showTools and (flags["tooloutline"] or false)
    local tOutColor   = flags["toolcoloroutline"] and flags["toolcoloroutline"].Color or Color3.fromRGB(0, 0, 0)

    local showDist    = flags["distance"] or false
    local distColor   = flags["distcolor"]       and flags["distcolor"].Color       or Color3.fromRGB(255, 255, 255)
    local showDOutline = showDist and (flags["distoutline"] or false)
    local dOutColor   = flags["distcoloroutline"] and flags["distcoloroutline"].Color or Color3.fromRGB(0, 0, 0)

    local showHbar    = flags["healthbar"] or false
    local showHBOutline = showHbar and (flags["healthbaroutline"] or false)
    local hbOutColor  = flags["healthbarcoloroutline"] and flags["healthbarcoloroutline"].Color or Color3.fromRGB(0, 0, 0)
    local hbBarColor  = flags["healthbarcolor"] and flags["healthbarcolor"].Color or Color3.fromRGB(0, 255, 100)

    local camPos = getgenv().Camera.CFrame.Position

    if inclAcc ~= lastInclAcc then
        lastInclAcc = inclAcc
        for player in pairs(ESPObjects) do PartsCacheDirty[player] = true end
    end

    if flags["selfesp"] then
        if not ESPObjects[getgenv().LocalPlayer] then
            getgenv().AddPlayer(getgenv().LocalPlayer, true)
        end
    else
        if ESPObjects[getgenv().LocalPlayer] then
            getgenv().RemovePlayer(getgenv().LocalPlayer)
        end
    end

    for player, box in pairs(ESPObjects) do
        local character = player.Character
        if not character then getgenv().HideAll(player) continue end

        local hum = character:FindFirstChildOfClass("Humanoid")

        local isDead = false
        if hum then
            if hum:GetState() == Enum.HumanoidStateType.Dead then
                isDead = true
            elseif hum:GetAttribute("KO") == true or hum:GetAttribute("Knocked") == true then
                isDead = true
            end
        end
        if not isDead then
            if character:FindFirstChild("Dead") or character:FindFirstChild("KO") or character:FindFirstChild("Knocked") or character:FindFirstChild("SDeath") then
                isDead = true
            end
        end

        if isDead then
            getgenv().HideAll(player)
            continue
        end

        local root = character:FindFirstChild("HumanoidRootPart")
        if not root then getgenv().HideAll(player) continue end

        if PartsCacheDirty[player] then getgenv().BuildPartsCache(player) end
        local cache = PartsCache[player]
        if not cache then getgenv().HideAll(player) continue end

        local rootPos = root.Position
        local minX, minY =  math.huge,  math.huge
        local maxX, maxY = -math.huge, -math.huge
        local visible = false

        if sizeType == "fixed" then
            local corners = getgenv().GetFixedCorners(rootPos)
            for i = 1, 8 do
                local sp = getgenv().Camera:WorldToViewportPoint(corners[i])
                if sp.Z > 0 then
                    visible = true
                    if sp.X < minX then minX = sp.X end if sp.Y < minY then minY = sp.Y end
                    if sp.X > maxX then maxX = sp.X end if sp.Y > maxY then maxY = sp.Y end
                end
            end
        else
            local function processParts(parts)
                for _, part in ipairs(parts) do
                    if part and part.Parent and (part.Position - rootPos).Magnitude <= 10 then
                        local corners = getgenv().GetAllCorners(part)
                        for i = 1, 8 do
                            local sp = getgenv().Camera:WorldToViewportPoint(corners[i])
                            if sp.Z > 0 then
                                visible = true
                                if sp.X < minX then minX = sp.X end if sp.Y < minY then minY = sp.Y end
                                if sp.X > maxX then maxX = sp.X end if sp.Y > maxY then maxY = sp.Y end
                            end
                        end
                    end
                end
            end
            processParts(cache.all)
            if inclAcc then processParts(cache.accs) end
        end

        if not visible or (maxX - minX) > 500 or (maxY - minY) > 600 then
            getgenv().HideAll(player) continue
        end

        if showBox then
            if mode == "corner" then
                getgenv().DrawCorner(box, minX, minY, maxX, maxY, boxColor, outColor, true, showOutline, fillColor, fillTrans, showFill)
            else
                getgenv().Draw2D(box, minX, minY, maxX, maxY, boxColor, outColor, true, showOutline, fillColor, fillTrans, showFill)
            end
        else
            getgenv().HideBox(box)
        end

        local centerX   = (minX + maxX) * 0.5
        local dist      = (camPos - root.Position).Magnitude
        local scaledSize = math.clamp(math.floor(NAME_BASE_SIZE * (NAME_BASE_DIST / math.max(dist, 1))), NAME_MIN_SIZE, NAME_MAX_SIZE)

        local label = NameObjects[player]
        if label then
            if showNames then
                local nameStr = nameType == "username" and player.Name or player.DisplayName
                label.Text                  = nameStr
                label.TextColor3            = nameColor
                label.Font                  = nameFont
                label.TextSize              = scaledSize
                label.Position              = UDim2.fromOffset(centerX, minY - 2)
                label.TextStrokeColor3      = nOutColor
                label.TextStrokeTransparency = showNOutline and 0 or 1
                label.Visible               = true
            else
                label.Visible = false
            end
        end

        local belowY = maxY + 4

        local tlabel = ToolObjects[player]
        if tlabel then
            if showTools then
                local tool    = character:FindFirstChildOfClass("Tool")
                local toolStr = tool and tool.Name or "no tool"
                tlabel.Text                  = toolStr
                tlabel.TextColor3            = toolColor
                tlabel.Font                  = nameFont
                tlabel.TextSize              = scaledSize
                tlabel.Position              = UDim2.fromOffset(centerX, belowY)
                tlabel.TextStrokeColor3      = tOutColor
                tlabel.TextStrokeTransparency = showTOutline and 0 or 1
                tlabel.Visible               = true
                belowY = belowY + scaledSize + 2
            else
                tlabel.Visible = false
            end
        end

        local dlabel = DistObjects[player]
        if dlabel then
            if showDist then
                dlabel.Text                  = math.floor(dist) .. "m"
                dlabel.TextColor3            = distColor
                dlabel.Font                  = nameFont
                dlabel.TextSize              = scaledSize
                dlabel.Position              = UDim2.fromOffset(centerX, belowY)
                dlabel.TextStrokeColor3      = dOutColor
                dlabel.TextStrokeTransparency = showDOutline and 0 or 1
                dlabel.Visible               = true
            else
                dlabel.Visible = false
            end
        end

        local hb = HealthbarObjects[player]
        if hb and showHbar and hum then
            local pct = math.clamp(hum.Health / math.max(hum.MaxHealth, 1), 0, 1)
            local barH = math.max(maxY - minY, 1)
            local fillH = math.max(math.floor(barH * pct), 1)
            local barX = minX - HB_WIDTH - HB_OFFSET
            local barY = maxY - fillH

            hb.bar.Position = Vector2.new(barX, barY)
            hb.bar.Size = Vector2.new(HB_WIDTH, fillH)
            hb.bar.Color = hbBarColor
            hb.bar.Transparency = 1
            hb.bar.Visible = true

            if showHBOutline then
                hb.outline.Position = Vector2.new(barX - 1, barY - 1)
                hb.outline.Size = Vector2.new(HB_WIDTH + 2, fillH + 2)
                hb.outline.Color = hbOutColor
                hb.outline.Visible = true
            else
                hb.outline.Visible = false
            end
        elseif hb then
            getgenv().HideHealthbarDrawings(hb)
        end
    end
end)

local visible = false
local lastPos = nil

getgenv().RunService.RenderStepped:Connect(function(dt)
    local settings = getgenv().getSettings()
    if not settings.enabled then
        image.ImageTransparency = 1
        visible = false
        return
    end

    local character = getgenv().LocalPlayer.Character
    local anchor = character and (
        character:FindFirstChild("UpperTorso") or
        character:FindFirstChild("Torso") or
        character:FindFirstChild("HumanoidRootPart")
    )
    if not anchor then
        image.ImageTransparency = 1
        visible = false
        return
    end

    local worldPos = desync_api.get_desync_pos() or anchor.Position
    local pos, onScreen = getgenv().Camera:WorldToViewportPoint(worldPos)

    if not onScreen or pos.Z <= 0 then
        image.ImageTransparency = 1
        visible = false
        return
    end

    image.ImageTransparency = 0

    local speed = 1 + (19 * settings.lerp)
    local factor = math.min(dt * speed, 1)

    if not lastPos then
        lastPos = pos
    end

    local sx = lastPos.X + (pos.X - lastPos.X) * factor
    local sy = lastPos.Y + (pos.Y - lastPos.Y) * factor
    lastPos = Vector3.new(sx, sy, pos.Z)

    local size = settings.sizePx
    local iconAsset = ICON_ASSETS[settings.iconIdx + 1]

    if image.Image ~= iconAsset then
        image.Image = iconAsset
    end

    if image.Size.X.Offset ~= size then
        image.Size = UDim2.fromOffset(size, size)
    end

    getgenv().place(image, size, sx, sy)
end)

local drawings = {}

local height = 0.7
local radius = 2
local sides = 25
local hatTrs = 0.35
local lineTrs = 1
local speed = 0.2
local offsetY = 0.5

getgenv().lerp = function(a,b,t)
    return a + (b-a)*t
end

getgenv().lerpColor = function(a,b,t)
    return Color3.new(
        getgenv().lerp(a.R,b.R,t),
        getgenv().lerp(a.G,b.G,t),
        getgenv().lerp(a.B,b.B,t)
    )
end

getgenv().getColor = function(p,time)
    local c1 = getgenv().china_c1
    local c2 = getgenv().china_c2
    local c3 = getgenv().china_c3
    local c4 = getgenv().china_c4

    local t = getgenv().china_hat_spin and time * speed or 0
    local s = (p + t) % 1

    if s < 0.25 then
        return getgenv().lerpColor(c1,c2,s/0.25)
    elseif s < 0.5 then
        return getgenv().lerpColor(c2,c3,(s-0.25)/0.25)
    elseif s < 0.75 then
        return getgenv().lerpColor(c3,c4,(s-0.5)/0.25)
    else
        return getgenv().lerpColor(c4,c1,(s-0.75)/0.25)
    end
end

getgenv().createDrawings = function()
    if #drawings == sides then return end

    for _,d in ipairs(drawings) do
        pcall(function() d[1]:Remove() end)
        pcall(function() d[2]:Remove() end)
    end

    drawings = {}

    for _=1,sides do
        local l = Drawing.new("Line")
        local t = Drawing.new("Triangle")

        l.ZIndex = 2
        l.Thickness = 2

        t.ZIndex = 1
        t.Filled = true

        table.insert(drawings,{l,t})
    end
end

getgenv().RunService.RenderStepped:Connect(function()
    if not getgenv().china_hat_enabled then
        for _,d in ipairs(drawings) do
            d[1].Visible = false
            d[2].Visible = false
        end
        return
    end

    local char = getgenv().LocalPlayer.Character
    local head = char and char:FindFirstChild("Head")
    local hum = char and char:FindFirstChildOfClass("Humanoid")

    if not (char and head and hum and hum.Health > 0) then return end

    getgenv().createDrawings()

    local time = tick()
    local full = math.pi*2

    local top = head.Position + Vector3.new(0,offsetY + height,0)
    local base = head.Position + Vector3.new(0,offsetY,0)

    for i=1,sides do
        local line = drawings[i][1]
        local tri = drawings[i][2]

        local p1 = i/sides
        local p2 = (i%sides+1)/sides

        local a1 = p1*full
        local a2 = p2*full

        local pos1 = base + Vector3.new(math.cos(a1),0,math.sin(a1))*radius
        local pos2 = base + Vector3.new(math.cos(a2),0,math.sin(a2))*radius

        local s1,on1 = getgenv().Camera:WorldToViewportPoint(pos1)
        local s2,on2 = getgenv().Camera:WorldToViewportPoint(pos2)
        local sTop,onTop = getgenv().Camera:WorldToViewportPoint(top)

        if not (on1 and on2 and onTop) then
            line.Visible = false
            tri.Visible = false
            continue
        end

        local col = getgenv().getColor(p1,time)

        line.From = Vector2.new(s1.X,s1.Y)
        line.To = Vector2.new(s2.X,s2.Y)
        line.Color = col
        line.Transparency = lineTrs
        line.Visible = true

        tri.PointA = Vector2.new(sTop.X,sTop.Y)
        tri.PointB = line.From
        tri.PointC = line.To
        tri.Color = col
        tri.Transparency = hatTrs
        tri.Visible = true
    end
end)

local originalTextures = {}

getgenv().savePart = function(v)
    if not originalTextures[v] then
        originalTextures[v] = {
            material = v.Material,
            color = v.Color,
            texture = v:IsA("MeshPart") and v.TextureID or nil
        }
    end
end

getgenv().setPart = function(v, mat, color)
    getgenv().savePart(v)
    v.Material = mat
    v.Color = color or Color3.new(1, 1, 1)
    if v:IsA("MeshPart") then
        v.TextureID = ""
    end
end

getgenv().restorePart = function(v)
    local orig = originalTextures[v]
    if orig then
        v.Material = orig.material
        v.Color = orig.color
        if v:IsA("MeshPart") and orig.texture ~= nil then
            v.TextureID = orig.texture
        end
        originalTextures[v] = nil
    end
end

getgenv().isToolPart = function(v)
    local p = v.Parent
    while p do
        if p:IsA("Tool") then return true end
        p = p.Parent
    end
    return false
end

getgenv().applyChams = function(char)
    local mat = materialMap[getgenv().selfchams_material] or Enum.Material.Plastic
    local parts = {}
    for _, v in ipairs(char:GetDescendants()) do
        if v:IsA("BasePart") and not getgenv().isToolPart(v) then
            table.insert(parts, v)
        end
    end
    for i, v in ipairs(parts) do
        local color = i % 2 == 0 and getgenv().selfchams_color2 or getgenv().selfchams_color
        getgenv().setPart(v, mat, color)
    end
end

getgenv().resetChams = function(char)
    for _, v in ipairs(char:GetDescendants()) do
        if v:IsA("BasePart") and not getgenv().isToolPart(v) then
            getgenv().restorePart(v)
        end
    end
end

getgenv().updateCrosshairDimensions = function()

    parts.left.Size = UDim2.new(0, length, 0, thickness)

    parts.right.Position = UDim2.new(0.5, gap, 0.5, 0)
    parts.right.Size = UDim2.new(0, length, 0, thickness)

    parts.top.Position = UDim2.new(0.5, 0, 0.5, -gap - length)
    parts.top.Size = UDim2.new(0, thickness, 0, length)

    parts.bottom.Position = UDim2.new(0.5, 0, 0.5, gap)
    parts.bottom.Size = UDim2.new(0, thickness, 0, length)

    if parts.leftStroke then
        parts.leftStroke.Thickness = getgenv().crosshairOutline
    end
    if parts.rightStroke then
        parts.rightStroke.Thickness = getgenv().crosshairOutline
    end
    if parts.topStroke then
        parts.topStroke.Thickness = getgenv().crosshairOutline
    end
    if parts.bottomStroke then
        parts.bottomStroke.Thickness = getgenv().crosshairOutline
    end
end

getgenv().updateCrosshairPosition = function()
    local parts = getgenv().crosshairParts
    if not parts or not parts.holder then return end

    if getgenv().crosshairChase then
        local mouse = getgenv().LocalPlayer:GetMouse()
        parts.holder.Position = UDim2.new(0, mouse.X, 0, mouse.Y)
        if parts.indicatorHolder then
            parts.indicatorHolder.Position = UDim2.new(0, mouse.X, 0, mouse.Y + getgenv().crosshairGap + getgenv().crosshairLength + 6)
        end
    else
        parts.holder.Position = UDim2.new(0.5, 0, 0.5, -28)
        if parts.indicatorHolder then
            parts.indicatorHolder.Position = UDim2.new(0.5, 0, 0.5, -28 + getgenv().crosshairGap + getgenv().crosshairLength + 6)
        end
    end
end

getgenv().updateCrosshairRotation = function(deltaTime)
    local parts = getgenv().crosshairParts
    if not parts or not parts.holder then return end

    if getgenv().crosshairRotate then
        local step = getgenv().crosshairRotateSpeed * 60 * deltaTime
        getgenv().crosshairRotation = getgenv().crosshairRotation + step * getgenv().crosshairRotDir

        if getgenv().crosshairRotation >= 360 then
            getgenv().crosshairRotation = 360
            getgenv().crosshairRotDir = -1
        elseif getgenv().crosshairRotation <= 0 then
            getgenv().crosshairRotation = 0
            getgenv().crosshairRotDir = 1
        end

        parts.holder.Rotation = getgenv().crosshairRotation
    else
        parts.holder.Rotation = 0
        getgenv().crosshairRotation = 0
        getgenv().crosshairRotDir = 1
    end
end

getgenv().updateCrosshairIndicator = function()
    local parts = getgenv().crosshairParts

    parts.indicatorLabel.Visible = getgenv().crosshairIndicator

            ColorSequenceKeypoint.new(1, getgenv().crosshairIndicatorColor2)
        })
    end
end

local newPartConnection

getgenv().applyToPart = function(part)
    if not part:IsA('BasePart') or processedParts[part] then
        return
    end
    local texId = materials[part.Material.Name]
    if texId then
        for _, face in ipairs(faces) do
            local tex = Instance.new('Texture')
            tex.Name = 'TsunamiTex'
            tex.ZIndex = 2147483647
            tex.Texture = 'rbxassetid://' .. texId
            tex.Face = face
            tex.Color3 = part.Color
            tex.Transparency = part.Transparency
            tex.StudsPerTileU = 4
            tex.StudsPerTileV = 4
            tex.Parent = part
        end
        part.Material = Enum.Material.SmoothPlastic
        processedParts[part] = true
    else
    end
end

getgenv().applyTextures = function()
    for _, part in ipairs(getgenv().Workspace:GetDescendants()) do
        getgenv().applyToPart(part)
    end
end

getgenv().removeTextures = function()
    for _, obj in ipairs(getgenv().Workspace:GetDescendants()) do
        if obj:IsA('Texture') and obj.Name == 'TsunamiTex' then
            obj:Destroy()
        end
    end
    processedParts = {}
end

textureChanger.setEnabled = function(val)
    isEnabled = val
    if val then
        task.spawn(getgenv().applyTextures)
        if newPartConnection then newPartConnection:Disconnect() end
        newPartConnection = getgenv().Workspace.DescendantAdded:Connect(function(part)
            if isEnabled then
                getgenv().applyToPart(part)
            end
        end)
    else
        if newPartConnection then newPartConnection:Disconnect() end
        newPartConnection = nil
        task.spawn(getgenv().removeTextures)
    end
end

getgenv().applySkybox = function(name)
    for _, v in pairs(getgenv().Lighting:GetChildren()) do
        if v:IsA("Sky") then
            v:Destroy()
        end
    end

    if name ~= "default" then
        local data = predefinedSkyboxes[name]
        if data then
            local sky = Instance.new("Sky")
            sky.SkyboxBk = "rbxassetid://" .. data.Bk
            sky.SkyboxDn = "rbxassetid://" .. data.Dn
            sky.SkyboxFt = "rbxassetid://" .. data.Ft
            sky.SkyboxLf = "rbxassetid://" .. data.Lf
            sky.SkyboxRt = "rbxassetid://" .. data.Rt
            sky.SkyboxUp = "rbxassetid://" .. data.Up
            sky.Parent = getgenv().Lighting
        end
    end
end

getgenv().playWeatherSound = function(name)
    local id = getgenv().weatherSounds[name]
    if not id then return end
    
    if getgenv().currentWeatherSound == name then
        getgenv().stopWeatherSound()
        return
    end
    
    getgenv().stopWeatherSound()
    
    getgenv().weatherSound = Instance.new("Sound")
    getgenv().weatherSound.SoundId = "rbxassetid://" .. id
    getgenv().weatherSound.Looped = true
    getgenv().weatherSound.Volume = 10
    getgenv().weatherSound.Parent = getgenv().SoundService
    getgenv().weatherSound:Play()
    getgenv().currentWeatherSound = name
end

local WeatherTextures = {
    Rain = "rbxassetid://3806148993",
    Snow = "rbxassetid://917186750",
    Autumn = "rbxassetid://915916073"
}

getgenv().createWeather = function()
    local char = getgenv().LocalPlayer.Character or getgenv().LocalPlayer.CharacterAdded:Wait()
    local head = char:WaitForChild("Head")

    local part = Instance.new("Part")
    part.Size = Vector3.new(300, 300, 300)
    part.Transparency = 1
    part.CanCollide = false
    part.CanQuery = false
    part.CanTouch = false
    part.Anchored = false
    part.Massless = true
    part.Parent = getgenv().Workspace

    local weld = Instance.new("WeldConstraint")
    weld.Part0 = head
    weld.Part1 = part
    weld.Parent = part

    local emitter = Instance.new("ParticleEmitter")
    emitter.Parent = part

    if getgenv().currentWeather == "rain" then
        emitter.Texture = WeatherTextures.Rain
        emitter.Rate = 15000
        emitter.Speed = NumberRange.new(200)
        emitter.Lifetime = NumberRange.new(2.5)
        emitter.Size = NumberSequence.new(5)
        emitter.EmissionDirection = Enum.NormalId.Bottom

    elseif getgenv().currentWeather == "snow" then
        emitter.Texture = WeatherTextures.Snow
        emitter.Rate = 500
        emitter.Speed = NumberRange.new(15)
        emitter.Size = NumberSequence.new({
            NumberSequenceKeypoint.new(0, 0.33),
            NumberSequenceKeypoint.new(0.5, 0.4),
            NumberSequenceKeypoint.new(1, 0.33)
        })
        emitter.EmissionDirection = Enum.NormalId.Bottom
        emitter.SpreadAngle = Vector2.new(50, 50)

    elseif getgenv().currentWeather == "autumn" then
        emitter.Texture = WeatherTextures.Autumn
        emitter.Rate = 1000
        emitter.Speed = NumberRange.new(30)
        emitter.Size = NumberSequence.new(0.8)
        emitter.Orientation = Enum.ParticleOrientation.VelocityPerpendicular
        emitter.EmissionDirection = Enum.NormalId.Bottom
        emitter.SpreadAngle = Vector2.new(50, 50)
    end

    return part
end

getgenv().startWeather = function()
    if getgenv().weatherPart then
        getgenv().weatherPart:Destroy()
        getgenv().weatherPart = nil
    end
    getgenv().weatherPart = getgenv().createWeather()
end

getgenv().stopWeather = function()
    if getgenv().weatherPart then
        getgenv().weatherPart:Destroy()
        getgenv().weatherPart = nil
    end
end

getgenv().restartWeather = function()
    if getgenv().weatherEnabled then
        getgenv().stopWeather()
        task.wait()
        getgenv().startWeather()
    end
end

getgenv().createCore = function(character)
    if getgenv().corePart and getgenv().corePart.Parent then
        getgenv().corePart:Destroy()
    end
    local core = Instance.new("Part")
    core.Name = "FlightCore"
    core.Size = Vector3.new(0.05, 0.05, 0.05)
    core.CanCollide = false
    core.Transparency = 1
    core.Parent = getgenv().Workspace
    local weld = Instance.new("Weld", core)
    weld.Part0 = core
    weld.Part1 = character and character:FindFirstChild("HumanoidRootPart")
    weld.C0 = CFrame.new(0, 0, 0)
    getgenv().corePart = core
    return core
end

getgenv().flySpeed = 500

getgenv().startFly = function()
    if getgenv().flying then return end
    local character = getgenv().getCharacter()
    if not character then return end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not humanoid then return end
    getgenv().flying = true
    humanoid.PlatformStand = true
    local core = getgenv().createCore(character)
    local bv = Instance.new("BodyVelocity", core)
    bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
    bv.Velocity = Vector3.zero
    local bg = Instance.new("BodyGyro", core)
    bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bg.P = 9e4
    bg.CFrame = core.CFrame
    local conn = getgenv().RunService.Heartbeat:Connect(function()
        if not getgenv().flying then return end
        local focused = getgenv().UserInputService:GetFocusedTextBox()
        if focused then return end
        local camera = getgenv().Camera
        local moveDir = Vector3.zero
        if getgenv().UserInputService:IsKeyDown(Enum.KeyCode.W) then moveDir = moveDir + camera.CFrame.LookVector end
        if getgenv().UserInputService:IsKeyDown(Enum.KeyCode.S) then moveDir = moveDir - camera.CFrame.LookVector end
        if getgenv().UserInputService:IsKeyDown(Enum.KeyCode.A) then moveDir = moveDir - camera.CFrame.RightVector end
        if getgenv().UserInputService:IsKeyDown(Enum.KeyCode.D) then moveDir = moveDir + camera.CFrame.RightVector end
        if getgenv().UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveDir = moveDir + Vector3.new(0, 1, 0) end
        if getgenv().UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then moveDir = moveDir - Vector3.new(0, 1, 0) end
        bv.Velocity = moveDir * getgenv().flySpeed
        bg.CFrame = camera.CFrame
    end)
    table.insert(getgenv().flightConnections, conn)
end

getgenv().stopFly = function()
    getgenv().flying = false
    for _, conn in ipairs(getgenv().flightConnections) do
        conn:Disconnect()
    end
    getgenv().flightConnections = {}
    local character = getgenv().getCharacter()
    if character then
        local humanoid = character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            humanoid.PlatformStand = false
        end
    end
    if getgenv().corePart then
        getgenv().corePart:Destroy()
        getgenv().corePart = nil
    end
end

getgenv().findClosestPlayer = function(searchText)
    if not searchText or searchText == "" then
        return #getgenv().playerList > 0 and getgenv().playerList[1] or nil
    end
    local lower = searchText:lower()
    local bestMatch = nil
    local bestScore = -1
    for _, name in ipairs(getgenv().playerList) do
        local player = getgenv().Players:FindFirstChild(name)
        if not player then continue end
        local nameLower = name:lower()
        local displayLower = player.DisplayName:lower()
        if nameLower == lower or displayLower == lower then
            return name
        end
        if nameLower:find(lower, 1, true) then
            local score = #lower / #nameLower
            if score > bestScore then
                bestScore = score
                bestMatch = name
            end
        elseif displayLower:find(lower, 1, true) then
            local score = #lower / #displayLower
            if score > bestScore then
                bestScore = score
                bestMatch = name
            end
        end
    end
    return bestMatch or (#getgenv().playerList > 0 and getgenv().playerList[1] or nil)
end

getgenv().normalize = function(str)
    return tostring(str or ""):lower():gsub("%s+", ""):gsub("[%[%]]","")
end

getgenv().parseNumber = function(v)
    if not v then return 0 end
    local s = tostring(v)
    local digits = s:match("%d+")
    return tonumber(digits) or 0
end

getgenv().findInventoryAmmoObject = function(inventory, ammoShopName)
    if not inventory then return nil end
    local ammoSearch = tostring(ammoShopName):gsub("^%s*%d+%s*", "")
    local normSearch = getgenv().normalize(ammoSearch)
    for _, child in ipairs(inventory:GetChildren()) do
        if getgenv().normalize(child.Name):find(normSearch, 1, true) or normSearch:find(getgenv().normalize(child.Name), 1, true) then
            return child
        end
    end
    return nil
end

getgenv().getSpareAmmoValue = function(inventory, ammoShopName)
    local obj = getgenv().findInventoryAmmoObject(inventory, ammoShopName)
    if not obj then return 0, nil end
    local val = nil
    if obj.Value ~= nil then
        val = getgenv().parseNumber(obj.Value)
    elseif tonumber(obj.Name) then
        val = getgenv().parseNumber(obj.Name)
    else
        val = 0
    end
    return val, obj
end

local lastReloadVariant = "Reload"
local reloadInputConnection = nil
local reloadDetected = false

getgenv().setAutoReload = function(enabled)
    getgenv().autoReloadEnabled = enabled

    if enabled and not getgenv().autoReloadConnection then
        getgenv().autoReloadConnection = getgenv().RunService.Heartbeat:Connect(function()
            if not getgenv().autoReloadEnabled then
                return
            end

            local mainEvent = getgenv().getMainEvent()
            local char = getgenv().LocalPlayer.Character

            if not char then
                return
            end

            for _, obj in ipairs(char:GetChildren()) do
                if obj:IsA("Tool") then
                    local ammo = obj:FindFirstChild("Ammo")

                    if ammo and ammo.Value <= 0 then
                        if mainEvent then
                            pcall(function()
                                mainEvent:FireServer(lastReloadVariant, obj)
                            end)
                        else
                            pcall(function()
                                getgenv().UserInputService:SendKeyEvent(true, Enum.KeyCode.R, false)
                                task.wait(0.05)
                                getgenv().UserInputService:SendKeyEvent(false, Enum.KeyCode.R, false)
                            end)
                        end
                    end
                end
            end
        end)
    elseif not enabled and getgenv().autoReloadConnection then
        getgenv().autoReloadConnection:Disconnect()
        getgenv().autoReloadConnection = nil
    end
    
    if enabled and not reloadInputConnection then
        reloadInputConnection = getgenv().UserInputService.InputBegan:Connect(function(input, gameProcessed)
            if gameProcessed then return end
            if input.KeyCode == Enum.KeyCode.R then
                local char = getgenv().LocalPlayer.Character
                if not char then return end
                
                for _, tool in ipairs(char:GetChildren()) do
                    if tool:IsA("Tool") then
                        local ammo = tool:FindFirstChild("Ammo")
                        if ammo and ammo.Value <= 0 then
                            pcall(function()
                                local mainEvent = getgenv().getMainEvent()
                                if mainEvent then
                                    mainEvent:FireServer(lastReloadVariant, tool)
                                end
                            end)
                        end
                    end
                end
            end
        end)
    elseif not enabled and reloadInputConnection then
        reloadInputConnection:Disconnect()
        reloadInputConnection = nil
    end
end

getgenv().setReloadVariant = function(variant)
    lastReloadVariant = variant
end

local function autoDetectReloadVariant()
    local char = getgenv().LocalPlayer.Character
    if not char then return end
    
    for _, tool in ipairs(char:GetChildren()) do
        if tool:IsA("Tool") then
            local ammo = tool:FindFirstChild("Ammo")
            if ammo and ammo.Value <= 0 then
                for _, variant in ipairs(reloadVariations) do
                    pcall(function()
                        local mainEvent = getgenv().getMainEvent()
                        if mainEvent then
                            mainEvent:FireServer(variant, tool)
                            lastReloadVariant = variant
                            reloadDetected = true
                        end
                    end)
                end
                return
            end
        end
    end
end

getgenv().getAnimationFromAsset = function(id)
    local success, objects = pcall(function()
        return game:GetObjects("rbxassetid://" .. id)
    end)

    if success and objects then
        for _, obj in ipairs(objects) do
            if obj:IsA("Animation") then
                return obj
            end

            local found = obj:FindFirstChildWhichIsA("Animation", true)
            if found then
                return found
            end
        end
    end

    return nil
end

getgenv().updateEmoteSpeed = function(speed)
    local n = tonumber(speed) or 1
    getgenv().emoteSpeed = n

    if library and library.flags then
        library.flags.emotes_speed = n
    end

    if getgenv().currentEmoteTrack then
        pcall(function()
            getgenv().currentEmoteTrack:AdjustSpeed(n)
        end)
    end
end

getgenv().startEmote = function()
    if getgenv().emotePlaying then
        return
    end

    local char = getgenv().LocalPlayer.Character
    if not char then
        return
    end

    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then
        return
    end

    local animator = hum:FindFirstChildOfClass("Animator")
    if not animator then
        animator = Instance.new("Animator")
        animator.Parent = hum
    end

    local input = getgenv().selectedEmote == "custom" and getgenv().customEmoteId or presetEmotes[getgenv().selectedEmote]
    if not input or input == "" then
        return
    end

    local animId = input:match("%d+")
    if not animId then
        return
    end

    local animation = getgenv().getAnimationFromAsset(animId)
    if not animation then
        animation = Instance.new("Animation")
        animation.AnimationId = "rbxassetid://" .. animId
    end

    local success, track = pcall(function()
        return animator:LoadAnimation(animation)
    end)

    if not success or not track then
        warn("Failed to load animation:", animId)
        return
    end

    local speed = tonumber((library and library.flags and library.flags.emotes_speed) or getgenv().emoteSpeed) or 1

    track.Priority = Enum.AnimationPriority.Action
    track.Looped = true
    track:Play(0, 1, speed)
    track:AdjustSpeed(speed)

    getgenv().currentEmoteTrack = track
    getgenv().emotePlaying = true

    track.Stopped:Connect(function()
        if getgenv().currentEmoteTrack == track then
            getgenv().currentEmoteTrack = nil
            getgenv().emotePlaying = false
        end

        pcall(function()
            if animation then
                animation:Destroy()
            end
        end)
    end)
end

getgenv().stopEmote = function()

        pcall(function()
            getgenv().currentEmoteTrack:Stop()
        end)
        getgenv().currentEmoteTrack = nil
    end
    getgenv().emotePlaying = false
end

getgenv().MAX_ANGULAR_VEL = 30

getgenv().startAntiFling = function()
    if getgenv().antiFlingConnection then return end
    
    getgenv().antiFlingConnection = getgenv().RunService.RenderStepped:Connect(function()
        if not getgenv().antiFlingEnabled then return end
        
        local myChar = getgenv().LocalPlayer.Character
        if not myChar then return end
        
        local myHum = myChar:FindFirstChildOfClass("Humanoid")
        local myRoot = myChar:FindFirstChild("HumanoidRootPart")
        
        if not (myHum and myRoot and myHum.Health > 0) then return end
        
        for _, plr in ipairs(getgenv().Players:GetPlayers()) do
            if plr ~= getgenv().LocalPlayer then
                local char = plr.Character
                
                if char then
                    local hrp = char:FindFirstChild("HumanoidRootPart")
                    
                    if hrp then
                        if hrp.Velocity.Magnitude > getgenv().MAX_VELOCITY then
                            hrp.Velocity = Vector3.zero
                        end
                        
                        if hrp.RotVelocity.Magnitude > getgenv().MAX_ANGULAR_VEL then
                            hrp.RotVelocity = Vector3.zero
                        end
                        
                        if hrp.Size.X > 5 or hrp.Size.Y > 5 or hrp.Size.Z > 5 then
                            hrp.Size = Vector3.new(2, 2, 1)
                        end
                        
                        if hrp.CanCollide == false then
                            hrp.CanCollide = true
                        end
                    end
                end
            end
        end
    end)
end

getgenv().localSounds = {}
local supportedExts = {wav=true, mp3=true, ogg=true, mp4=true}

getgenv().scanSounds = function()
    local found = {}
    local ok, files = pcall(lf, "void/sounds")
    if ok and files then
        for _, path in ipairs(files) do
            local ext = path:match("%.(%w+)$")
            if ext and supportedExts[ext:lower()] then
                local name = path:match("([^/\\]+)$") or path
                found[name] = path
            end
        end
    end
    return found
end

getgenv().localSounds = getgenv().scanSounds()

getgenv().rebuildNames = function()
    local names = {}
    for k in pairs(getgenv().hitsoundIds) do
        table.insert(names, k)
    end
    for k in pairs(getgenv().localSounds) do
        table.insert(names, k)
    end
    table.sort(names)
    return names
end

getgenv().resolveSoundId = function(name)
    local localPath = getgenv().localSounds[name]
    if localPath then
        if not isf(localPath) then return nil end
        if getcustomasset then return getcustomasset(localPath) else return nil end
    end
    return getgenv().hitsoundIds[name]
end

getgenv().playHitSound = function()
    if not library.flags.hits_enable then return end
    local name = library.flags.hitsound_selected
    if not name then return end
    local id = getgenv().resolveSoundId(name)
    if not id then return end
    local volume = library.flags.hitsounds_volume or 10
    local sound = Instance.new("Sound")
    sound.SoundId = id
    sound.Volume = volume
    sound.Parent = getgenv().SoundService
    sound:Play()
    sound.Ended:Connect(function() sound:Destroy() end)
    task.delay(5, function() if sound and sound.Parent then sound:Destroy() end end)
end

getgenv().isPlayerSelectedAsTarget = function(player)
    local targets = getgenv().selectedTargets
    if not targets then return false end
    if type(targets) == "table" then
        for _, name in ipairs(targets) do
            if name == player.Name then return true end
        end
    elseif type(targets) == "string" then
        return targets == player.Name
    end
    return false
end

end

getgenv().trackPlayerHealth = function(player)
    if getgenv().healthConnections[player] then
        pcall(function() getgenv().healthConnections[player]:Disconnect() end)
        getgenv().healthConnections[player] = nil
    end

    local function onHealthChanged(currentHP)
        local soundEnabled = library.flags.hits_enable
        local notifyEnabled = library.flags.hitn_enable
        if not soundEnabled and not notifyEnabled then return end

        local activeTool = getgenv().getHeldTool()
        if not activeTool then return end

        local isRageTarget = false
        if getgenv().ragebotEnabled then
            isRageTarget = getgenv().isPlayerSelectedAsTarget(player)
        end

        local isSilentTarget = false
        local silentPlayer = getgenv().getSilentAimTargetPlayer()
        if silentPlayer and silentPlayer == player then
            isSilentTarget = true
        end

        if not (isRageTarget or isSilentTarget) then return end

        local lastHP = getgenv().lastHealthCache[player]
        if lastHP ~= nil and currentHP < lastHP then
            local dmg = lastHP - currentHP
            if soundEnabled then getgenv().playHitSound() end
            if notifyEnabled then
                getgenv().sendHitNotify({
                    player = player.Name,
                    tool = activeTool.Name,
                    dmg = math.floor(dmg)
                })
            end
        end
        getgenv().lastHealthCache[player] = currentHP
    end

    local function setupHumanoid(character)
        task.wait(0.1)
        local hum = character:FindFirstChildOfClass("Humanoid")
        if not hum then return end
        getgenv().lastHealthCache[player] = hum.Health
        local conn = hum.HealthChanged:Connect(onHealthChanged)
        getgenv().healthConnections[player] = conn
    end

    if player.Character then
        setupHumanoid(player.Character)
    end

    player.CharacterAdded:Connect(function(character)
        getgenv().lastHealthCache[player] = nil
        if getgenv().healthConnections[player] then
            pcall(function() getgenv().healthConnections[player]:Disconnect() end)
            getgenv().healthConnections[player] = nil
        end
        setupHumanoid(character)
    end)
end

getgenv().enforcePremiumValues = function()
    if not getgenv().fakePremiumEnabled then return end
    
    local player = getgenv().LocalPlayer
    local dataFolder = player:FindFirstChild("DataFolder")
    if not dataFolder then return end
    
    local subscription = dataFolder:FindFirstChild("Subscription")
    if not subscription then return end
    
    local hasSubscription = subscription:FindFirstChild("HasSubscription")
    local isPremium = subscription:FindFirstChild("IsPremium")
    local subscriptionData = subscription:FindFirstChild("SubscriptionData")
    local subscriptionStreak = subscription:FindFirstChild("SubscriptionStreak")
    local subscriptionStreakExpiry = subscription:FindFirstChild("SubscriptionStreakExpiry")
    
    if hasSubscription and hasSubscription:IsA("BoolValue") then
        hasSubscription.Value = true
    end
    
    if isPremium and isPremium:IsA("BoolValue") then
        isPremium.Value = true
    end
    
    if subscriptionData and (subscriptionData:IsA("NumberValue") or subscriptionData:IsA("IntValue")) then
        subscriptionData.Value = math.huge
    end
    
    if subscriptionStreak and (subscriptionStreak:IsA("NumberValue") or subscriptionStreak:IsA("IntValue")) then
        subscriptionStreak.Value = math.huge
    end
    
    if subscriptionStreakExpiry and (subscriptionStreakExpiry:IsA("NumberValue") or subscriptionStreakExpiry:IsA("IntValue")) then
        subscriptionStreakExpiry.Value = math.huge
    end
end

getgenv().setupPremiumHooks = function()
    if getgenv().premiumConnection then
        getgenv().premiumConnection:Disconnect()
    end
    
    local player = getgenv().LocalPlayer
    local dataFolder = player:FindFirstChild("DataFolder")
    if not dataFolder then return end
    
    local subscription = dataFolder:FindFirstChild("Subscription")
    if not subscription then return end
    
    for _, child in pairs(subscription:GetChildren()) do
        if child:IsA("IntValue") or child:IsA("NumberValue") or child:IsA("BoolValue") then
            local conn = child.Changed:Connect(function()
                getgenv().enforcePremiumValues()
            end)
            if not getgenv().premiumConnection then
                getgenv().premiumConnection = conn
            end
        end
    end
    
    getgenv().enforcePremiumValues()
end

getgenv().startAnimationBreaker = function(char)
    getgenv().disconnectAnimConnections()
    if not char then return end
    local hum = char:FindFirstChildOfClass("Humanoid")
    if not hum then return end

    getgenv().animationConnections.RenderStepped = getgenv().RunService.RenderStepped:Connect(function()
        if not getgenv().AnimationBreakerEnabled then return end
        local currentChar = getgenv().GetCharacter2()
        if not currentChar then return end
        local currentHum = currentChar:FindFirstChildOfClass("Humanoid")
        if not currentHum then return end

        for _, track in pairs(currentHum:GetPlayingAnimationTracks()) do
            if track.IsPlaying then
                track:AdjustSpeed(getgenv().LagAmount / 50)
                if getgenv().JitterAmount > 0 then
                    local jitterOffset = (math.random() - 0.5) * (getgenv().JitterAmount / 100)
                    local newPos = track.TimePosition + jitterOffset
                    if newPos >= 0 and newPos <= track.Length then
                        track.TimePosition = newPos
                    end
                end
            end
        end
    end)

    getgenv().animationConnections.Heartbeat = getgenv().RunService.Heartbeat:Connect(function()
        if not getgenv().AnimationBreakerEnabled then return end
        local currentChar = getgenv().GetCharacter2()
        if not currentChar then return end
        local currentHum = currentChar:FindFirstChildOfClass("Humanoid")
        if not currentHum then return end

        for _, track in pairs(currentHum:GetPlayingAnimationTracks()) do
            if track.IsPlaying then
                local speedVariation = (getgenv().LagAmount / 50) + ((math.random() - 0.5) * (getgenv().JitterAmount / 100))
                track:AdjustSpeed(math.max(0.1, speedVariation))
            end
        end
    end)
end

getgenv().stopAnimationBreaker = function()
    getgenv().disconnectAnimConnections()
    local char = getgenv().GetCharacter2()
    if char then
        local hum = char:FindFirstChildOfClass("Humanoid")
        if hum then
            for _, track in pairs(hum:GetPlayingAnimationTracks()) do
                pcall(function() track:AdjustSpeed(1) end)
            end
        end
    end
end
